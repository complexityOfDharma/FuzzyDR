FUNCTION_BLOCK agent 

VAR_INPUT
    selfEnergy: REAL;
    neighborEnergy: REAL;
END_VAR

VAR_OUTPUT
    agreement: REAL;
END_VAR

FUZZIFY selfEnergy
    TERM low: TRAPE 0 0 3 5;
    TERM ok: TRAPE 3 5 7 9;
    TERM great: TRAPE 7 9 10 10;
END_FUZZIFY

FUZZIFY neighborEnergy
    TERM low: TRAPE 0 0 3 5;
    TERM ok: TRAPE 3 5 7 9;
    TERM great: TRAPE 7 9 10 10;
END_FUZZIFY

DEFUZZIFY agreement
    TERM strongly_agree: TRAPE 0 0 0.5 0.7;
    TERM oppose: TRAPE 0.3 0.5 1 1;
    METHOD: COG; // Center of Gravity
END_DEFUZZIFY

RULEBLOCK No1
    // Add rules as needed. For example:
    RULE 1: IF selfEnergy IS great AND neighborEnergy IS great THEN agreement IS strongly_agree;
    RULE 2: IF selfEnergy IS low THEN agreement IS oppose;
END_RULEBLOCK

END_FUNCTION_BLOCK
